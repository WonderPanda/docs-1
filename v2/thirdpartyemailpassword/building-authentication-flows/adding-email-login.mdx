### Custom UI

<CustomUITabs>

<TabItem value="web">

<FrontendWebSubTabs>

<TabItem value="npm">

After collecting user information, you can call the `emailPasswordSignUp` function:

```tsx
import ThirdPartyEmailPassword from "supertokens-web-js/recipe/thirdpartyemailpassword";

async function signup(email: string, password: string) {
    const signUpResponse = await ThirdPartyEmailPassword.emailPasswordSignUp({
        formFields: [
            {
                id: "email",
                value: email,
            },
            {
                id: "password",
                value: password,
            }
        ],
    });

    if (signUpResponse.status === "OK") {
        // Success, you can access the user information
        const user = signUpResponse.user;
    }

    // Error handling
    if (signUpResponse.status === "FIELD_ERROR") {
        // Iterate through form fields to identify the error
        signUpResponse.formFields.forEach(field => {
            // The id will match the values you passed when calling signUp
            console.log("Error in field with id:", field.id)
            console.log(field.error);
        })
    }
}
```

</TabItem>
